function ajax(url) {
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    xhr.open("GET", url, true);
    xhr.responseType = "blob";
    xhr.onload = () => {
      if (xhr.readyState == 4) {
        if (xhr.status == 200) {
          const blob = xhr.response;
          resolve(blob);
        } else {
          reject(xhr.statusText);
        }
      }
    };
  });
}
